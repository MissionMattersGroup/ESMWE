    /*
    * Description: Method to to get Progress Note Details
    * Input      : Id(string) of Progress Note
    * Output     : List of Progress Note
    */
	public List<Force__Progress_Note__c> getProgressNote(string strProgressNoteId) 
	{
    	    //Start --Progress Note Query Record Fetch and Create LIst of FIleds to display 
            string queryStr="SELECT";
            string fixFieldsProgressNote=" Id,Service__c,Family_Signature__c,Staff_Signature__c,Supervisor_Signature__c,IsProgressNoteRecordLocked__c,";
            fixFieldsProgressNote += "IsProgressNoteRecordLockedForSupervisor__c,Family_Signee_Name__c,Staff_Signee_Name__c,Supervisor_Signee_Name__c,";
            fixFieldsProgressNote += "Family_Signed_Date__c,Staff_Signed_Date__c,Supervisor_Signed_Date__c,";
            fixFieldsProgressNote += "External_Signature__c,External_Signee_Name__c,External_Role_Name__c,External_Relationship__c,External_Signed_Date__c,";
            fixFieldsProgressNote += "Family_Role_Name__c,Staff_Role_Name__c,Supervisor_Role_Name__c,Relationship_Type_Family__c";
            queryStr+= createQueryStringFields(fixFieldsProgressNote,DataBag.strProgressNoteFields,"Force__Progress_Note__c");
            queryStr+=" WHERE Id=\""+strProgressNoteId+"\"";
            queryStr+=" LIMIT 1";
        
        //Assign the dynamic query to the Progress Note 
        return  (List<Force__Progress_Note__c>)Database.Query(queryStr);
	}
    
	/*
    * Description: Method to save the progress note with the updated values
    * Input      : lstProgressNoteObj : lst of Progress Note which have the updated values
    * Output     : Redirects to the Page with updated values
    */
   
    [HttpPost]
	public ActionResponse SaveProgressNote(List<Force__Progress_Note__c> lstProgressNoteObj){
	    int count = 0;
       
        foreach (var progressNoteObj in lstProgressNoteObj)
        {
            foreach (var field in DataBag.lstProgressNoteFields)
            {
                //To save the progress note data when opens in new view
                if(DataBag.strProgressNoteTypeView == "new") {
                    if(field != "Staff_Name__c") {
                        string val = Request["m$" + count.ToString() + "$." +  field];
                        progressNoteObj.Set(field, val);  
                    }
                    if(field == "Staff_Name__c") {
                        progressNoteObj.Set("Staff_Name__c", DataBag.strStaffContactId);
                    }
                }
                //To save the progress note data when opens in edit view
                if (DataBag.strProgressNoteTypeView == "edit") {
                    if(field != "Staff_Name__c" && field != "Status__c"){
                        string val = Request["m$" + count.ToString() + "$." +  field];
                        progressNoteObj.Set(field, val);  
                    }
                }
                //To save the status as in progress when click on save in detail and edit mode
                if (DataBag.strProgressNoteTypeView == "edit" || DataBag.strProgressNoteTypeView == "detail") {
                    if(field == "Status__c"){
                        progressNoteObj.Set("Status__c", "In Progress");  
                    }
                }   
            }
             count++;
        }
        
        // On save check if progress note id is present then update the record otherwise create the new record of progress note
        if (lstProgressNoteObj.Count > 0) {
            if(string.IsNullOrEmpty(lstProgressNoteObj[0].Id)){
                try {
                    Insert(lstProgressNoteObj);
                    DataBag.Exception = "";
                } catch (Exception e) {
                    DataBag.Exception = e.Message;
                    if(DataBag.IsMultipleMethodThrowException == true){
	                     throw e;
	                }
                    return View();
                }
                DataBag.strProgressNoteId = lstProgressNoteObj[0].Id;
            }
            else {
                try {
                    Edit(lstProgressNoteObj);
                    DataBag.Exception = "";
                } catch (Exception e) {
                    DataBag.Exception = e.Message;
                    if(DataBag.IsMultipleMethodThrowException == true){
                         DataBag.IsMultipleMethodThrowException = false;
	                     throw e;
	            }
                    return View();
                }
            }
        }
        return Redirect(ActivePages.ProgressNoteInterface, new { typeView = "detail",id=DataBag.strProgressNoteId });
    }

	/*
    * Description: Method to save the signature on Progress Note object for Staff and Supervisor Signature 
    * Input      : lstProgressNoteObj : list of Progress Note which have the updated values
    * Output     : save the signatures
    */
    [HttpPost]
    public ActionResponse SaveSignature(List<Force__Progress_Note__c> lstProgressNoteObj){
        
        foreach (var progressNoteObj in lstProgressNoteObj)
        {
            //Stored the value of Staff signature
            string StaffSignval    = Request["StaffSignToUpdateProgressNote"];
            
            //Check whether the Staff Signature is Blank or Not
            bool checkStaffSignatureVal = checkTheSignatureValue(StaffSignval);
            
            //Stored the value of Supervisor signature
            string SupervisorSignval = Request["SupervisorSignToUpdateProgressNote"];
           
            //Check whether the Supervisor Signature is Blank or Not
            bool checkSupervisorSignatureVal = checkTheSignatureValue(SupervisorSignval);
            
            //stored the value of family signee name
            string StaffSigneeName = Request["m$0$.Staff_Signee_Name__c"]; 
            
            //stored the value of family signee name
            string SupervisorSigneeName = Request["m$0$.Supervisor_Signee_Name__c"];
            
            //Stored the value of External signature
            string ExternalSignval = Request["ExternalSignToUpdateProgressNote"];
            
            //Check whether the External Signature is Blank or Not
            bool checkExternalSignatureVal = checkTheSignatureValue(ExternalSignval);
            
            //stored the value of External signee name
            string ExternalName = Request["m$0$.External_Signee_Name__c"]; 
            
            //stored the value of External signee's relationship to the client
            string ExternalRelationship = Request["m$0$.External_Relationship__c"]; 
            
            if(checkStaffSignatureVal == false){
                progressNoteObj.Set("Staff_Signature__c", "");
                progressNoteObj.Set("Staff_Signed_Date__c","");
                progressNoteObj.Set("Staff_Signee_Name__c", "");
                progressNoteObj.Set("Staff_Role_Name__c", "");
            }
            else{
                // To check if the existing and new signatures are not same then change the date and signee's title
                if(progressNoteObj.Staff_Signature__c != StaffSignval) {
                    progressNoteObj.Set("Staff_Signed_Date__c", DateTime.Today);
                    progressNoteObj.Set("Staff_Role_Name__c", UserInfo.Role.Name);
                }
                progressNoteObj.Set("Staff_Signature__c", StaffSignval);
                progressNoteObj.Set("Staff_Signee_Name__c", StaffSigneeName);
            }
            
            if(checkSupervisorSignatureVal == false){

	/*
    * Description: Method to check the value of the signature is blank or not
    * Input      : signatureVal: The signature for which the value needs to be checked
    * Output     : boolean true: if it the signature is blank and false: if not
    */
    public bool checkTheSignatureValue(string signatureVal){
        int firstIndexValues         = signatureVal.IndexOf('[');
        int lastIndexValues          = signatureVal.LastIndexOf(']');
        int lengthOfStrSupervisorSig = lastIndexValues - firstIndexValues + 1;
    
        string subStringOfSignVal = signatureVal.Substring(firstIndexValues+1,lengthOfStrSupervisorSig-2);
        if(!string.IsNullOrEmpty(subStringOfSignVal)){
            return true;
        }
        return false;
    }

	public string createQueryStringFields(string fixFields, string strOfQueryFields, string ObjectName){
	    
	    string strQuery = "";

                                              .Select(f => new { f.Id,f.Funder__c,f.Client_Service__c,f.Status__c,f.Billing_Code__c})
                                              .ToListAsAdmin();             
	     
	    getFunderOption(lstAuthorizationLineitems);
	    
	    return  getBillingCodeOption(lstAuthorizationLineitems);
	  
	}
    
    /*
    * Description: Method to get funder related to Autnoriztion's funder
    * Input :lstAuthorizationLineitems of Authorization Line item
    */
	public void getFunderOption(List<Force__Authorization_Line_Item__c> lstAuthorizationLineitems){
	    
	    List<Force__Funder__c> lstFunder = new List<Force__Funder__c>();
	    
	    List<string> setOfFunderId = new List<string>();
	     
	     if(lstAuthorizationLineitems.Count > 0) { 
	        
            foreach(var objAuthorization in lstAuthorizationLineitems) {
                
                //check funder is present in authorization line items list
                if(objAuthorization.Funder__c != "") {
                    setOfFunderId.Add(objAuthorization.Funder__c);
                }
    	    }
            
            if(setOfFunderId.Count > 0) {
                
                //List to query funder related to Autnoriztion's funder
                lstFunder = Database.Query<Force__Funder__c>(f => setOfFunderId.Contains(f.Id))
                                              .Select(f => new { f.Id,f.Name })
                                              .ToListAsAdmin(); 
                
                List<WrapperForOption> lstFunderOptions = new List<WrapperForOption>();
    			    
			    foreach(var objFunder in lstFunder) {
		            WrapperForOption wrapperInstance = new WrapperForOption();
		            
		            //Wrapper to set key and values of the Funder
		            wrapperInstance.key = (string) objFunder.Get("Id");
    			    wrapperInstance.value = (string) objFunder.Get("Name");
    			    lstFunderOptions.Add(wrapperInstance);
		        }
		        //list of databag with funder options
        	    DataBag.lstFunderOptions = lstFunderOptions;
            }
	    } 
	}
	
	/*
    * Description: Method to get Billing code related to Autnoriztion's funder
    * Input :lstAuthorizationLineitems of Authorization Line item
    */
	public Dictionary<string,List<WrapperForOption>> getBillingCodeOption(List<Force__Authorization_Line_Item__c> lstAuthorizationLineitems ){
	    
	    List<Force__Funder_Code__c> lstFunderCode = new List<Force__Funder_Code__c>();
	    
	    List<string> setOfFunderCodeId = new List<string>();
	     
	    mapFunderId_lstFunderCode = new Dictionary<string,List<WrapperForOption>>();
	    
	    Dictionary<string,List<string>> mapFunderCodeId_lstFunderId = new Dictionary<string,List<string>>();
	    
	    // To check the size of Authorization Line items list
	    if(lstAuthorizationLineitems.Count > 0) { 
	        
            foreach(var objAuthorization in lstAuthorizationLineitems) {
                
                // To check if billing code is present in authorization line items
                if(objAuthorization.Billing_Code__c != "") {
                    
                    // To check if funder is present in authorization line items
                    if(objAuthorization.Funder__c != "") {
                      
                        //Map to fill with Billing Code Id and list of Funder if map not contains the billing code
                        if(!mapFunderCodeId_lstFunderId.ContainsKey(objAuthorization.Billing_Code__c)) {
                            
                            List<string> lstFunderId =  new List<string>();
                            lstFunderId.Add(objAuthorization.Funder__c);
                            
                            //Add Billing code and funder list to the map
                            mapFunderCodeId_lstFunderId.Add(objAuthorization.Billing_Code__c,lstFunderId);
                            
                            //Map to fill with Funder Code Id and list of Billing code with new instance
                            if(!mapFunderId_lstFunderCode.ContainsKey(objAuthorization.Funder__c)){
                                mapFunderId_lstFunderCode.Add(objAuthorization.Funder__c,new List<WrapperForOption>()) ; 
                            }
                        }
                        else{
                            List<string> lstFunderId = new List<string>();
                            //To get the list of funder
                            lstFunderId = mapFunderCodeId_lstFunderId[objAuthorization.Billing_Code__c];
                            
                            //Override the map values with list of funder
                            if (!lstFunderId.Contains(objAuthorization.Funder__c)) {
                                lstFunderId.Add(objAuthorization.Funder__c);
                                mapFunderCodeId_lstFunderId[objAuthorization.Billing_Code__c]= lstFunderId;
                            }
                        }
                    }
                }
    	    }
            
            //Check if funder code ids count are greater than zero
            if(mapFunderCodeId_lstFunderId.Keys.Count > 0) {
                
                //List to query funder code related to Autnoriztion's funder
                lstFunderCode = Database.Query<Force__Funder_Code__c>(f => mapFunderCodeId_lstFunderId.Keys.Contains(f.Id))
                                              .Select(f => new { f.Id,f.Name })
                                              .ToListAsAdmin(); 
                
                List<WrapperForOption> lstFunderCodeOptions = new List<WrapperForOption>();
    			    
			    foreach(var objFunderCode in lstFunderCode) {
		            WrapperForOption wrapperInstance = new WrapperForOption();
		            //Wrapper to set key and values of the billing code
		            wrapperInstance.key = (string) objFunderCode.Get("Id");
    			    wrapperInstance.value = (string) objFunderCode.Get("Name");
    			    
    			    //check map contains the key(billing code id)
    			    if(mapFunderCodeId_lstFunderId.ContainsKey(wrapperInstance.key)) {
        			    foreach(var strFunderId in mapFunderCodeId_lstFunderId[wrapperInstance.key]) {
        			    
            			    List<WrapperForOption> lstFunderCodeItermidiateOption = mapFunderId_lstFunderCode[strFunderId];
            			    lstFunderCodeItermidiateOption.Add(wrapperInstance);
            			    
            			    //override map with list of billing code
            			    mapFunderId_lstFunderCode[strFunderId]= lstFunderCodeItermidiateOption;
        			    }
    			    }
		        }
            }
	    }
	    
	    return mapFunderId_lstFunderCode;
	}
